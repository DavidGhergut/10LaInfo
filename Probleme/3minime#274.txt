Se dau n numere naturale. Determina≈£i cele mai mici trei numere dintre cele date.

SOLUTIE:

#include <iostream>

using namespace std;

int main(){
    int n, v[1005], i, min1, min2, min3;
    cin >> n;
    for (i = 0; i < n; i++)
        cin >> v[i];
    min1 = 10000;
    min2 = 10000;
    min3 = 10000;
    for (i = 0; i < n; i++){
        if (v[i] < min1){
             min3 = min2;
             min2 = min1;
             min1 = v[i];
        }
        else if (v[i] < min2){
            min3 = min2;
            min2 = v[i];
        }
        else if (v[i] < min3)
            min3 = v[i];
    }
    cout << min3 << " " << min2 << " " << min1;
    return 0;
}
